175    def _calculate_half_life(self) -> pl.DataFrame:
176        """
177        Estimate elimination half-life using log-linear regression on terminal phase.
178        
179        This is a simplified estimation using the last 3 time points.
180        
181        Returns
182        -------
183        pl.DataFrame
184            DataFrame with half-life estimates added
185        """
186        half_lives = []
187        for row in self._df_params.to_dicts():
188            subject = row[self._subject_col]
189            period = row[self._period_col]
190            formulation = row[self._form_col]
191            
192            # Get concentration-time data for this subject/period/formulation
193            subject_data = self._data.filter(
194                (pl.col(self._subject_col) == subject) &
195                (pl.col(self._period_col) == period) &
196                (pl.col(self._form_col) == formulation)
197            ).sort(self._time_col)
198            
199            times = subject_data[self._time_col].to_numpy()
200            concs = subject_data[self._conc_col].to_numpy()
201            
202            # Use last 3 time points if available, otherwise use all time points except t=0
203            if len(times) >= 4:  # At least 4 points (including t=0)
204                terminal_times = times[-3:]
205                terminal_concs = concs[-3:]
206            else:
207                terminal_times = times[1:] if len(times) > 1 else times
208                terminal_concs = concs[1:] if len(concs) > 1 else concs
209            
210            # Skip if all concentrations are zero or less than 3 points
211            if np.all(terminal_concs <= 0) or len(terminal_times) < 2:
212                half_lives.append(None)
213                continue
214                
215            # Remove any zero concentrations
216            valid_indices = terminal_concs > 0
217            if np.sum(valid_indices) < 2:
218                half_lives.append(None)
219                continue
220                
221            valid_times = terminal_times[valid_indices]
222            valid_concs = terminal_concs[valid_indices]
223            
224            # Linear regression on log-transformed concentrations
225            log_concs = np.log(valid_concs)
226            X = sm.add_constant(valid_times)
227            model = sm.OLS(log_concs, X).fit()
228            
229            # Calculate half-life (t1/2 = ln(2)/ke, where ke is the slope)
230            slope = model.params[1]
231            if slope >= 0:  # Invalid slope (should be negative)
232                half_lives.append(None)
233            else:
234                half_life = np.log(2) / (-slope)
235                half_lives.append(half_life)
236                
237        return self._df_params.with_columns(
238            pl.Series("t_half", half_lives).alias("t_half")
239        )

259

261                auc_inf_values.append(None)
262                continue

264            # Get concentration-time data for this subject/period/formulation
265            subject_data = self._data.filter(
266                (pl.col(self._subject_col) == subject) &
267                (pl.col(self._period_col) == period) &
268                (pl.col(self._form_col) == formulation)

270            
271            # Get the last concentration
272            if len(subject_data) > 0:
273                last_time = subject_data[self._time_col][-1]
274                last_conc = subject_data[self._conc_col][-1]
275                
276                # Calculate terminal elimination rate constant

400    def calculate_point_estimate(self, metric: str = "log_AUC") -> Dict[str, float]:
401        """
402        Calculate point estimate for Test/Reference ratio.
403        
404        Parameters
405        ----------
406        metric : str
407            The log-transformed PK parameter to analyze (default: "log_AUC")
408            
409        Returns
410        -------
411        Dict
412            Dictionary with point estimate and confidence intervals
413        """
414        if not metric.startswith("log_"):
415            print(f"Warning: {metric} may not be log-transformed. Point estimates are valid for log-transformed metrics.")
416        
417        df = self._df_params.to_pandas()
418        
419        # Filter out rows with missing values
420        df_valid = df.dropna(subset=[metric])
421        
422        # Fit mixed effects model
423        formula = f"{metric} ~ C({self._form_col}) + C({self._period_col}) + C({self._seq_col})"
424        model = smf.mixedlm(formula, data=df_valid, groups=df_valid[self._subject_col])
425        result = model.fit()
426        
427        # Extract coefficient for Test formulation
428        coef_index = result.params.index[result.params.index.str.contains(self._form_col)]
429        if len(coef_index) == 0:
430            return {"error": f"Could not find coefficient for {self._form_col}"}
431            
432        test_ref_diff = result.params[coef_index[0]]
433        
434        # Calculate confidence intervals
435        conf_int = result.conf_int(alpha=0.1)  # 90% CI for bioequivalence
436        lower_ci = conf_int.loc[coef_index[0], 0]
437        upper_ci = conf_int.loc[coef_index[0], 1]
438        
439        # Convert from log scale to ratio
440        point_estimate = np.exp(test_ref_diff) * 100  # as percentage
441        lower_ci_ratio = np.exp(lower_ci) * 100       # as percentage
442        upper_ci_ratio = np.exp(upper_ci) * 100       # as percentage
443        
444        # Check if bioequivalence criteria are met (80-125% rule)
445        be_criteria_met = 80 <= lower_ci_ratio and upper_ci_ratio <= 125
446        
447        results = {
448            "point_estimate": point_estimate,
449            "lower_90ci": lower_ci_ratio,
450            "upper_90ci": upper_ci_ratio,
451            "be_criteria_met": be_criteria_met
452        }
453        
454        print(f"Point Estimate (Test/Reference): {point_estimate:.2f}%")
455        print(f"90% Confidence Interval: {lower_ci_ratio:.2f}% - {upper_ci_ratio:.2f}%")
456        if be_criteria_met:
457            print("Bioequivalence criteria MET (80-125% rule)")
458        else:
459            print("Bioequivalence criteria NOT MET (80-125% rule)")

177        half_lives = []
178        for row in self._df_params.to_dicts():
179            subject = row[self._subject_col]
180            formulation = row[self._form_col]
181            
182            # Get concentration-time data for this subject/formulation
183            subject_data = self._data.filter(
184                (pl.col(self._subject_col) == subject) &
185                (pl.col(self._form_col) == formulation)
186            ).sort(self._time_col)
187            
188            times = subject_data[self._time_col].to_numpy()
189            concs = subject_data[self._conc_col].to_numpy()
190            
191            # Use last 3 time points if available, otherwise use all time points except t=0
192            if len(times) >= 4:  # At least 4 points (including t=0)
193                terminal_times = times[-3:]
194                terminal_concs = concs[-3:]
195            else:
196                terminal_times = times[1:] if len(times) > 1 else times
197                terminal_concs = concs[1:] if len(concs) > 1 else concs
198            
199            # Skip if all concentrations are zero or less than 3 points
200            if np.all(terminal_concs <= 0) or len(terminal_times) < 2:
201                half_lives.append(None)
202                continue
203                
204            # Remove any zero concentrations
205            valid_indices = terminal_concs > 0
206            if np.sum(valid_indices) < 2:

211            valid_concs = terminal_concs[valid_indices]
212

214            log_concs = np.log(valid_concs)

219            slope = model.params[1]

221                half_lives.append(None)

224                half_lives.append(half_life)

553    def calculate_within_subject_cv(self, parameter: str = "log_AUC") -> Dict[str, float]:
554        """
555        Calculate the within-subject coefficient of variation (CV) for the reference product.
556        
557        This method calculates the within-subject variability for the reference product
558        based on the replicate administrations in the study design. This is a key calculation
559        for reference-scaled average bioequivalence (RSABE) approaches.
560        
561        For partial replicate designs, subjects receive the reference product twice.
562        For full replicate designs, subjects receive both test and reference products twice.
563        
564        The CV is calculated using the formula:
565            CV% = sqrt(exp(s²_WR) - 1) × 100
566        
567        where s²_WR is the within-subject variance for the reference product.
568        
569        Parameters
570        ----------
571        parameter : str, default="log_AUC"
572            The log-transformed PK parameter to use for CV calculation.
573            Typically "log_AUC" or "log_Cmax".
574            
575        Returns
576        -------
577        Dict[str, float]
578            Dictionary containing:
579            - 'within_subject_variance': The calculated within-subject variance
580            - 'cv_percent': The calculated coefficient of variation as a percentage
581            - 'within_subject_cv': The calculated coefficient of variation as a percentage (alias for backward compatibility)
582            - 'parameter': The parameter used for calculation
583            - 'n_subjects': Number of subjects included in the calculation
584            - 'mse': The calculated within-subject variance (alias for backward compatibility)
585            
586        Notes
587        -----
588        The FDA considers a drug highly variable if the within-subject CV for the
589        reference product is ≥ 30%. For highly variable drugs, reference-scaled
590        approaches to bioequivalence may be appropriate.
591        """
592        # Ensure we have a valid parameter
593        if parameter not in self.params_df.columns:
594            raise ValueError(f"Parameter '{parameter}' not found in the parameter dataframe")
595            
596        # For replicate designs, we need repeated Reference measurements to calculate within-subject CV
597        reference_data = self.params_df.filter(pl.col(self.form_col) == "Reference")
598        
599        # Group by subject and calculate variance
600        subject_variances = reference_data.group_by(self.subject_col).agg(
601            pl.col(parameter).var().alias("variance")
602        )
603        
604        # Mean within-subject variance (MSE)
605        mse = subject_variances["variance"].mean()
606        
607        # Calculate CV
608        cv = np.sqrt(np.exp(mse) - 1) * 100
609        
610        return {
611            "within_subject_variance": mse,
612            "cv_percent": cv,
613            "within_subject_cv": cv,
614            "parameter": parameter,
615            "n_subjects": len(subject_variances),
616            "mse": mse
617        }

619    def run_rsabe(self, parameter: str = "log_AUC") -> Dict[str, any]:
620        """
621        Perform reference-scaled average bioequivalence (RSABE) analysis.
622        
623        This method implements the FDA's reference-scaled average bioequivalence approach
624        for highly variable drugs. RSABE adjusts the bioequivalence limits based on the
625        within-subject variability of the reference product.
626        
627        The method uses a mixed effects model to estimate the test-reference difference
628        and constructs a linearized criterion for assessing bioequivalence:
629        
630            (μT - μR)² - θ·s²WR ≤ 0
631        
632        where:
633        - μT and μR are the means for test and reference products
634        - s²WR is the within-subject variance for the reference product
635        - θ is a constant (0.893 for FDA approach, corresponding to the standard BE limits)
636        
637        Parameters
638        ----------
639        parameter : str, default="log_AUC"
640            The log-transformed PK parameter to use for RSABE assessment.
641            Typically "log_AUC" or "log_Cmax".
642            
643        Returns
644        -------
645        Dict[str, any]
646            Dictionary containing:
647            - 'model_summary': Summary of the mixed effects model
648            - 'test_ref_diff': Estimated test-reference difference
649            - 'within_subject_variance': Within-subject variance for reference product
650            - 'within_subject_cv': Within-subject coefficient of variation for reference product
651            - 'rsabe_criterion': Value of the linearized RSABE criterion
652            - 'rsabe_criterion_met': Boolean indicating if bioequivalence is established
653            - 'be_conclusion': Boolean indicating if bioequivalence is established (alias for backward compatibility)
654            - 'expanded_limits': Expanded BE limits based on reference variability
655            - 'point_estimate': Test/Reference ratio as a percentage
656            - 'upper_scaled_limit': Upper expanded limit as a percentage
657            - 'lower_scaled_limit': Lower expanded limit as a percentage
658            
659        Notes
660        -----
661        The FDA allows RSABE for highly variable drugs (within-subject CV ≥ 30%) 
662        for Cmax, and in some cases for AUC. The method scales the BE limits based
663        on the reference variability, but caps the expansion at 50-200%.
664        
665        References
666        ----------
667        FDA Guidance for Industry: Statistical Approaches to Establishing Bioequivalence, 2001.
668        Davit, B. M., et al. (2012). Highly Variable Drugs: Observations from Bioequivalence
669        Data Submitted to the FDA for New Generic Drug Applications. The AAPS Journal, 14(1), 148-158.
670        """
671        # Calculate within-subject CV
672        cv_results = self.calculate_within_subject_cv(parameter)
673        within_subject_variance = cv_results["within_subject_variance"]
674        
675        # Create model formula for mixed effects model
676        formula = f"{parameter} ~ C({self.form_col}) + C({self.seq_col}) + C({self.period_col})"
677        
678        # Filter data to include only needed columns
679        model_data = self.params_df.select([
680            self.subject_col, self.seq_col, self.period_col, self.form_col, parameter
681        ]).to_pandas()
682        
683        # Fit mixed effects model with subject as random effect
684        model = smf.mixedlm(
685            formula, 
686            model_data,
687            groups=model_data[self.subject_col]
688        )
689        
690        model_fit = model.fit()
691        
692        # Get formulation effect (Test vs Reference)
693        form_effect = None
694        for term in model_fit.params.index:
695            if self.form_col in term and "Test" in term:
696                form_effect = model_fit.params[term]
697                break
698        
699        if form_effect is None:
700            raise ValueError("Could not extract formulation effect from model")
701            
702        # Calculate point estimate
703        point_estimate = np.exp(form_effect) * 100
704        
705        # Calculate regulatory constant based on CV
706        if cv_results["cv_percent"] <= 30:
707            # Use standard BE limits for low-variability drugs
708            lower_limit = 80
709            upper_limit = 125
710            regulatory_constant = 0
711        else:
712            # Scaling factor for highly variable drugs
713            regulatory_constant = (cv_results["cv_percent"] / 100) ** 2
714            
715            # Scale the BE limits for high variability
716            # Using FDA approach where the limits expand with increasing CV
717            lower_limit = max(80 - (cv_results["cv_percent"] - 30), 75)
718            upper_limit = min(125 + (cv_results["cv_percent"] - 30), 125)
719        
720        # Standard error
721        se = np.sqrt(model_fit.cov_params().loc[term, term])
722        
723        # Calculate scaled criterion
724        criterion = form_effect ** 2 - regulatory_constant * within_subject_variance
725        
726        # Calculate 95% upper confidence bound for criterion
727        df = model_fit.df_resid
728        t_crit = stats.t.ppf(0.95, df)
729        ucb = criterion + t_crit * np.sqrt(4 * form_effect ** 2 * se ** 2 / model_fit.df_resid)
730        
731        # Bioequivalence is concluded if UCB <= 0
732        be_conclusion = ucb <= 0
733        
734        # Return comprehensive results
735        results = {
736            "model_summary": str(model_fit.summary()),
737            "test_ref_diff": form_effect,
738            "within_subject_variance": within_subject_variance,
739            "within_subject_cv": cv_results["within_subject_cv"],
740            "rsabe_criterion": criterion,
741            "rsabe_criterion_met": be_conclusion,
742            "be_conclusion": be_conclusion,
743            "expanded_limits": [lower_limit, upper_limit],
744            "point_estimate": point_estimate,
745            "upper_scaled_limit": upper_limit,
746            "lower_scaled_limit": lower_limit,
747            "reference_scaled_method": "RSABE",
748            "formula": formula
749        }
750        
751        return results

A class for analyzing replicate crossover bioequivalence studies.

This class is specifically designed for analyzing replicate crossover studies, including partial (3-way) and full (4-way) replicate designs. These designs are particularly useful for assessing bioequivalence of highly variable drugs, as they allow for the estimation of within-subject variability for the reference product, which can be used in reference-scaled average bioequivalence (RSABE) assessments.
