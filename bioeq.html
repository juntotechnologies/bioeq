102    def _calculate_auc(self) -> pl.DataFrame:
103        """
104        Compute AUC (Area Under the Curve) using the trapezoidal rule.
105        
106        Returns
107        -------
108        pl.DataFrame
109            DataFrame with AUC values added
110        """
111        grouped_df = self._data.group_by(
112            [self._subject_col, self._period_col, self._seq_col, self._form_col]
113        ).agg([pl.col(self._time_col), pl.col(self._conc_col)])
114        auc_vals = [
115            np.trapezoid(row[self._conc_col], row[self._time_col])
116            for row in grouped_df.to_dicts()
117        ]
118        return grouped_df.with_columns(pl.Series("AUC", auc_vals))

175    def _calculate_half_life(self) -> pl.DataFrame:
176        """
177        Estimate elimination half-life using log-linear regression on terminal phase.
178        
179        This is a simplified estimation using the last 3 time points.
180        
181        Returns
182        -------
183        pl.DataFrame
184            DataFrame with half-life estimates added
185        """
186        half_lives = []
187        for row in self._df_params.to_dicts():
188            subject = row[self._subject_col]
189            period = row[self._period_col]
190            formulation = row[self._form_col]
191            
192            # Get concentration-time data for this subject/period/formulation
193            subject_data = self._data.filter(
194                (pl.col(self._subject_col) == subject) &
195                (pl.col(self._period_col) == period) &
196                (pl.col(self._form_col) == formulation)
197            ).sort(self._time_col)
198            
199            times = subject_data[self._time_col].to_numpy()
200            concs = subject_data[self._conc_col].to_numpy()
201            
202            # Use last 3 time points if available, otherwise use all time points except t=0
203            if len(times) >= 4:  # At least 4 points (including t=0)
204                terminal_times = times[-3:]
205                terminal_concs = concs[-3:]
206            else:
207                terminal_times = times[1:] if len(times) > 1 else times
208                terminal_concs = concs[1:] if len(concs) > 1 else concs
209            
210            # Skip if all concentrations are zero or less than 3 points
211            if np.all(terminal_concs <= 0) or len(terminal_times) < 2:
212                half_lives.append(None)
213                continue
214                
215            # Remove any zero concentrations
216            valid_indices = terminal_concs > 0
217            if np.sum(valid_indices) < 2:
218                half_lives.append(None)
219                continue
220                
221            valid_times = terminal_times[valid_indices]
222            valid_concs = terminal_concs[valid_indices]
223            
224            # Linear regression on log-transformed concentrations
225            log_concs = np.log(valid_concs)
226            X = sm.add_constant(valid_times)
227            model = sm.OLS(log_concs, X).fit()
228            
229            # Calculate half-life (t1/2 = ln(2)/ke, where ke is the slope)
230            slope = model.params[1]
231            if slope >= 0:  # Invalid slope (should be negative)
232                half_lives.append(None)
233            else:
234                half_life = np.log(2) / (-slope)
235                half_lives.append(half_life)
236                
237        return self._df_params.with_columns(
238            pl.Series("t_half", half_lives).alias("t_half")
239        )

241    def _calculate_auc_extrapolated(self) -> pl.DataFrame:
242        """
243        Calculate AUC extrapolated to infinity using the terminal elimination rate constant.
244        
245        AUC_inf = AUC_last + C_last/ke, where ke is obtained from half-life estimation.
246        
247        Returns
248        -------
249        pl.DataFrame
250            DataFrame with AUC_inf values added
251        """
252        auc_inf_values = []
253        
254        for row in self._df_params.to_dicts():
255            subject = row[self._subject_col]
256            period = row[self._period_col]
257            formulation = row[self._form_col]
258            t_half = row.get("t_half")
259            
260            if t_half is None or t_half <= 0:
261                auc_inf_values.append(None)
262                continue
263            
264            # Get concentration-time data for this subject/period/formulation
265            subject_data = self._data.filter(
266                (pl.col(self._subject_col) == subject) &
267                (pl.col(self._period_col) == period) &
268                (pl.col(self._form_col) == formulation)
269            ).sort(self._time_col)
270            
271            # Get the last concentration
272            if len(subject_data) > 0:
273                last_time = subject_data[self._time_col][-1]
274                last_conc = subject_data[self._conc_col][-1]
275                
276                # Calculate terminal elimination rate constant
277                ke = np.log(2) / t_half
278                
279                # Calculate extrapolated AUC
280                auc_extrapolated = last_conc / ke if ke > 0 else 0
281                auc_inf = row["AUC"] + auc_extrapolated
282                auc_inf_values.append(auc_inf)
283            else:
284                auc_inf_values.append(None)
285        
286        return self._df_params.with_columns(
287            pl.Series("AUC_inf", auc_inf_values).alias("AUC_inf")
288        )

400    def calculate_point_estimate(self, metric: str = "log_AUC") -> Dict[str, float]:
401        """
402        Calculate point estimate for Test/Reference ratio.
403        
404        Parameters
405        ----------
406        metric : str
407            The log-transformed PK parameter to analyze (default: "log_AUC")
408            
409        Returns
410        -------
411        Dict
412            Dictionary with point estimate and confidence intervals
413        """
414        if not metric.startswith("log_"):
415            print(f"Warning: {metric} may not be log-transformed. Point estimates are valid for log-transformed metrics.")
416        
417        df = self._df_params.to_pandas()
418        
419        # Filter out rows with missing values
420        df_valid = df.dropna(subset=[metric])
421        
422        # Fit mixed effects model
423        formula = f"{metric} ~ C({self._form_col}) + C({self._period_col}) + C({self._seq_col})"
424        model = smf.mixedlm(formula, data=df_valid, groups=df_valid[self._subject_col])
425        result = model.fit()
426        
427        # Extract coefficient for Test formulation
428        coef_index = result.params.index[result.params.index.str.contains(self._form_col)]
429        if len(coef_index) == 0:
430            return {"error": f"Could not find coefficient for {self._form_col}"}
431            
432        test_ref_diff = result.params[coef_index[0]]
433        
434        # Calculate confidence intervals
435        conf_int = result.conf_int(alpha=0.1)  # 90% CI for bioequivalence
436        lower_ci = conf_int.loc[coef_index[0], 0]
437        upper_ci = conf_int.loc[coef_index[0], 1]
438        
439        # Convert from log scale to ratio
440        point_estimate = np.exp(test_ref_diff) * 100  # as percentage
441        lower_ci_ratio = np.exp(lower_ci) * 100       # as percentage
442        upper_ci_ratio = np.exp(upper_ci) * 100       # as percentage
443        
444        # Check if bioequivalence criteria are met (80-125% rule)
445        be_criteria_met = 80 <= lower_ci_ratio and upper_ci_ratio <= 125
446        
447        results = {
448            "point_estimate": point_estimate,
449            "lower_90ci": lower_ci_ratio,
450            "upper_90ci": upper_ci_ratio,
451            "be_criteria_met": be_criteria_met
452        }
453        
454        print(f"Point Estimate (Test/Reference): {point_estimate:.2f}%")
455        print(f"90% Confidence Interval: {lower_ci_ratio:.2f}% - {upper_ci_ratio:.2f}%")
456        if be_criteria_met:
457            print("Bioequivalence criteria MET (80-125% rule)")
458        else:
459            print("Bioequivalence criteria NOT MET (80-125% rule)")
460            
461        return results

463    def summarize_pk_parameters(self) -> pl.DataFrame:
464        """
465        Calculate summary statistics for PK parameters by formulation.
466        
467        Returns
468        -------
469        pl.DataFrame
470            DataFrame with summary statistics
471        """
472        # Define the parameters to summarize
473        pk_params = ["AUC", "Cmax", "Tmax", "t_half", "AUC_inf"]

475        # Create an empty list to hold the result rows

491                    pl.col(param).max().alias("Max"),
492                    pl.col(param).count().alias("N")
493                ])
494            )
495            
496            # Convert to rows for each parameter
497            for row in summary.to_dicts():
498                summary_row = {
499                    "Parameter": param,
500                    "Formulation": row[self._form_col],
501                    "N": row["N"],
502                    "Mean": row["Mean"],
503                    "SD": row["SD"],
504                    "Median": row["Median"],
505                    "Min": row["Min"],
506                    "Max": row["Max"],
507                    "CV%": (row["SD"] / row["Mean"] * 100) if row["Mean"] != 0 else None
508                }
509                summary_rows.append(summary_row)
510        
511        # Create DataFrame from the rows

166    def _calculate_half_life(self) -> pl.DataFrame:
167        """
168        Estimate elimination half-life using log-linear regression on terminal phase.
169        
170        This is a simplified estimation using the last 3 time points.
171        
172        Returns
173        -------
174        pl.DataFrame
175            DataFrame with half-life estimates added
176        """
177        half_lives = []
178        for row in self._df_params.to_dicts():
179            subject = row[self._subject_col]
180            formulation = row[self._form_col]
181            
182            # Get concentration-time data for this subject/formulation
183            subject_data = self._data.filter(
184                (pl.col(self._subject_col) == subject) &
185                (pl.col(self._form_col) == formulation)
186            ).sort(self._time_col)
187            
188            times = subject_data[self._time_col].to_numpy()
189            concs = subject_data[self._conc_col].to_numpy()
190            
191            # Use last 3 time points if available, otherwise use all time points except t=0
192            if len(times) >= 4:  # At least 4 points (including t=0)
193                terminal_times = times[-3:]
194                terminal_concs = concs[-3:]
195            else:
196                terminal_times = times[1:] if len(times) > 1 else times
197                terminal_concs = concs[1:] if len(concs) > 1 else concs
198            
199            # Skip if all concentrations are zero or less than 3 points
200            if np.all(terminal_concs <= 0) or len(terminal_times) < 2:
201                half_lives.append(None)
202                continue
203                
204            # Remove any zero concentrations
205            valid_indices = terminal_concs > 0
206            if np.sum(valid_indices) < 2:
207                half_lives.append(None)
208                continue
209                
210            valid_times = terminal_times[valid_indices]
211            valid_concs = terminal_concs[valid_indices]
212            
213            # Linear regression on log-transformed concentrations
214            log_concs = np.log(valid_concs)
215            X = sm.add_constant(valid_times)
216            model = sm.OLS(log_concs, X).fit()
217            
218            # Calculate half-life (t1/2 = ln(2)/ke, where ke is the slope)
219            slope = model.params[1]
220            if slope >= 0:  # Invalid slope (should be negative)
221                half_lives.append(None)
222            else:
223                half_life = np.log(2) / (-slope)
224                half_lives.append(half_life)
225                
226        return self._df_params.with_columns(
227            pl.Series("t_half", half_lives).alias("t_half")
228        )

233        
234        AUC_inf = AUC_last + C_last/ke, where ke is obtained from half-life estimation.
235

248            if t_half is None or t_half <= 0:

252            # Get concentration-time data for this subject/formulation
253            subject_data = self._data.filter(
254                (pl.col(self._subject_col) == subject) &

def calculate_point_estimate(self, metric: str = 'log_AUC') -> Dict[str, float]: 
364    def calculate_point_estimate(self, metric: str = "log_AUC") -> Dict[str, float]:
365        """
366        Calculate point estimate for Test/Reference ratio.
367        
368        Parameters
369        ----------
370        metric : str
371            The log-transformed PK parameter to analyze (default: "log_AUC")
372            
373        Returns
374        -------
375        Dict
376            Dictionary with point estimate and confidence intervals
377        """
378        if not metric.startswith("log_"):
379            print(f"Warning: {metric} may not be log-transformed. Point estimates are valid for log-transformed metrics.")
380        
381        df = self._df_params.to_pandas()
382        unique_form = df[self._form_col].unique()
383        
384        if len(unique_form) != 2:
385            return {"error": f"Exactly 2 formulation levels required, but found {len(unique_form)}."}
386        
387        # Filter out rows with missing values
388        df_valid = df.dropna(subset=[metric])
389        
390        # Fit model with formulation as the only factor
391        formula = f"{metric} ~ C({self._form_col})"
392        model = smf.ols(formula, data=df_valid).fit()
393        
394        # Extract coefficient for Test formulation
395        coef_index = model.params.index[model.params.index.str.contains(self._form_col)]
396        if len(coef_index) == 0:
397            return {"error": f"Could not find coefficient for {self._form_col}"}
398            
399        test_ref_diff = model.params[coef_index[0]]
400        
401        # Calculate confidence intervals
402        conf_int = model.conf_int(alpha=0.1)  # 90% CI for bioequivalence
403        lower_ci = conf_int.loc[coef_index[0], 0]
404        upper_ci = conf_int.loc[coef_index[0], 1]
405        
406        # Convert from log scale to ratio
407        point_estimate = np.exp(test_ref_diff) * 100  # as percentage
408        lower_ci_ratio = np.exp(lower_ci) * 100       # as percentage
409        upper_ci_ratio = np.exp(upper_ci) * 100       # as percentage
410        
411        # Check if bioequivalence criteria are met (80-125% rule)
412        be_criteria_met = 80 <= lower_ci_ratio and upper_ci_ratio <= 125
413        
414        results = {
415            "point_estimate": point_estimate,
416            "lower_90ci": lower_ci_ratio,
417            "upper_90ci": upper_ci_ratio,
418            "be_criteria_met": be_criteria_met
419        }
420        
421        print(f"Point Estimate (Test/Reference): {point_estimate:.2f}%")
422        print(f"90% Confidence Interval: {lower_ci_ratio:.2f}% - {upper_ci_ratio:.2f}%")
423        if be_criteria_met:
424            print("Bioequivalence criteria MET (80-125% rule)")
425        else:
426            print("Bioequivalence criteria NOT MET (80-125% rule)")
427            
428        return results

430    def summarize_pk_parameters(self) -> pl.DataFrame:
431        """
432        Calculate summary statistics for PK parameters by formulation.
433        
434        Returns
435        -------
436        pl.DataFrame
437            DataFrame with summary statistics
438        """
439        # Define the parameters to summarize
440        pk_params = ["AUC", "Cmax", "Tmax", "t_half", "AUC_inf"]
441        
442        # Create an empty list to hold the result rows
443        summary_rows = []
444        
445        for param in pk_params:
446            if param not in self._df_params.columns:
447                continue
448                
449            # Group by formulation and calculate statistics
450            summary = (
451                self._df_params
452                .group_by(self._form_col)
453                .agg([
454                    pl.col(param).mean().alias("Mean"),
455                    pl.col(param).std().alias("SD"),
456                    pl.col(param).median().alias("Median"),
457                    pl.col(param).min().alias("Min"),
458                    pl.col(param).max().alias("Max"),
459                    pl.col(param).count().alias("N")
460                ])
461            )
462            
463            # Convert to rows for each parameter
464            for row in summary.to_dicts():
465                summary_row = {
466                    "Parameter": param,
467                    "Formulation": row[self._form_col],
468                    "N": row["N"],
469                    "Mean": row["Mean"],
470                    "SD": row["SD"],
471                    "Median": row["Median"],
472                    "Min": row["Min"],
473                    "Max": row["Max"],
474                    "CV%": (row["SD"] / row["Mean"] * 100) if row["Mean"] != 0 else None
475                }
476                summary_rows.append(summary_row)
477        
478        # Create DataFrame from the rows
479        return pl.DataFrame(summary_rows)
